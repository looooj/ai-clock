cmake_minimum_required(VERSION 3.10)

project(CountdownWxWidgets
    VERSION 1.0
    LANGUAGES CXX
)

# 指定 C++ 标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 查找 wxWidgets 包，至少需要 core 和 base 模块
find_package(wxWidgets REQUIRED COMPONENTS core base)

# 导入 wxWidgets 的 include 路径和编译选项
include(${wxWidgets_USE_FILE})

# 添加可执行目标
add_executable(Countdown
    main.cpp
)

# 链接 wxWidgets 库
target_link_libraries(Countdown
    ${wxWidgets_LIBRARIES}
)

# 可选：在 Windows 下启用 Unicode 支持
if (WIN32)
    target_compile_definitions(Countdown PRIVATE UNICODE _UNICODE)
endif()
